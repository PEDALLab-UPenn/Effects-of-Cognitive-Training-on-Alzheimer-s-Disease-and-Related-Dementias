	if a and b;
run;

	set active_claims_sdoh;

	reasscc = reasoning * scc_sdoh_comp;
	spdscc = speed * scc_sdoh_comp;

	ES_SDoH_comp = "Economic Stability SDoH Factor Score"
	AHC_SDoH_comp = "Access to Health Care SDoH Factor Score"

/********************CAUSE-SPECIFIC MODELS WITHOUT INTERACTION TERMS **************************************/
*THESE MODELS DO NOT INCLUDE RACE OR EDUCATION;

run;

	set output1 (rename=(col1=variance col2=estimate));
	hazard = exp(estimate);

	upper_cl = exp(estimate + 1.96*sqrt(variance));
